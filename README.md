```bash

npm install

# or using yarn
yarn install
```

## Available Commands

```bash
# Run the application
npm start

# Run in development mode
npm run dev

# Run tests
npm test

# Build the project
npm run build

# Lint the code
npm run lint

# Format the code
npm run format
```
